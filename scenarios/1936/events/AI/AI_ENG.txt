#	ENGAirAIGround		= Ground Support has priority

#	ENGInvadeItaly		= England is trying to invade Norway since low garrison
#	ENGInvadeNorway		= England is trying to invade Norway since low garrison

#	ENGHomeland			= England itself is being invaded

#	ENGNavalChannel		= Ignore the English Channel cause France fell
#	ENGNavalChannelP	= Protect the English Channel
#	ENGNavalNorway		= English Navy support Norway Invasion

#	ENGNavalChannelMed	= Same as Channel but shuts down Med

event = {
	id = 24000
	random = no 
	country = ENG
	
	trigger = {
		ai = yes
		war = { country = GER country = ENG }
		NOT = { war = { country = GER country = SOV } }
		OR = {
			control = { province = 55 data = -2 } # Paris held by enemy forces
			AND = {
				# Belgium
				control = { province = 114 data = -2 }
				control = { province = 115 data = -2 }
				control = { province = 117 data = -2 }
				control = { province = 116 data = -2 }
				control = { province = 113 data = -2 }
				control = { province = 108 data = -2 }
				control = { province = 110 data = -2 }
				control = { province = 111 data = -2 }
			}
		}
	}

	name = "AI_EVENT"
	style = 0	
	
	date = { day = 0 month = january year = 1936 }
	offset = 4
	deathdate = { day = 29 month = december year = 1963 }

	action_a = {
		command = { type = ai which = "switch/ENG_Britain.ai" }
	}
